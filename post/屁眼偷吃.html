Dataset的使用

# 这里是CatChord的Python哟

from torch.utils.data import Dataset
from PIL import Image
import os
# 设置类为Dataset
class MyData(Dataset):

    # 初始化
    def __init__(self, root_dir, label_dir):
        self.root_dir = root_dir
        self.label_dir = label_dir
        self.path = os.path.join(self.root_dir, self.label_dir)
        # 路径设置为该根路径下的子文件路径
        self.img_path = os.listdir(self.path)

    # 
    def __getitem__(self, idx):
        img_name = self.img_path[idx]
        img_item_path = os.path.join(self.root_dir, self.label_dir, img_name)
        img = Image.open(img_item_path)
        label = self.label_dir
        return img, label

    def __len__(self):
        return len(self.img_path)

root_dir = "dataset/train"
# 获得路径
cats_label_dir = "cats"
# 路径下的文件
cats_data = MyData(root_dir,cats_label_dir)
# 创建类
img, label = cats_data[0]
img.show()

基于MINIST手写体识别的学习

# 这里是CatChord的Python哟
# 数据科学与大数据技术二班
# 看看时间:2023/9/22 14:58
import torch
from torchvision.datasets import MNIST
from torchvision import transforms
from torch.utils.data import DataLoader
from torch import nn
import os
from PIL import Image
import numpy as np
# 预处理
transform = transforms.Compose([
    transforms.ToTensor(),   # 转化为tensor，范围为0-1
    transforms.Normalize((0.1307,), (0.3081))  # 归一化 (mean,std) 提高精度！
])

# 训练数据和测试数据
train_data = MNIST(root='./data', train=True, download=False, transform=transform)
train_loader = DataLoader(train_data, shuffle=True, batch_size=64)
test_data = MNIST(root='./data', train=False, download=True, transform=transform)
test_loader = DataLoader(test_data, shuffle=False, batch_size=64)
# print(train_data[0]) # 打印出来两个信息，一个是图片的信息，另一个是图片代表的数字
# train_data[1][0].show()

class Model(nn.Module): # 模型
    def __init__(self):
        super(Model, self).__init__()
        # 三层神经网络 784 --> 256 --> 64 --> 10
        # 定义一个神经网络的结构。这个网络包含三个线性层，也就是全连接层。
        self.linear1 = nn.Linear(784, 256)   # 28*28 = 784
        self.linear2 = nn.Linear(256, 64)
        self.linear3 = nn.Linear(64, 10)    # 10个输出

    def forward(self, x):
        x = x.view(-1, 784)     # 变形
        # 将输入数据x变形为一个二维张量，第二维的大小为784。-1表示第一维的大小由数据本身决定，这样可以处理任意批次大小的数据。
        # 通过第一层全连接层self.linear1处理数据，然后应用ReLU激活函数。ReLU函数将所有负值变为0，对正值不做改变。这样可以增加模型的非线性，帮助模型学习更复杂的特征。
        x = torch.relu(self.linear1(x))
        x = torch.relu(self.linear2(x))
        x = torch.relu(self.linear3(x))

        return x

model = Model()
criterion = nn.CrossEntropyLoss()   # 交叉熵损失 softmax + log + nllloss
optimizer = torch.optim.SGD(model.parameters(), 0.2)

# 加载预训练的模型参数
if os.path.exists('./model/model.pkl'):
   model.load_state_dict(torch.load("./model/model.pkl"))

def train(epoch):
    for index, data in enumerate(train_loader):
        input, target = data  # 输入和标签
        # 一行代码将优化器的梯度缓存清零。在PyTorch中，梯度是累积的，所以每次反向传播前都需要清零。
        optimizer.zero_grad()
        y_predict = model(input)	# 通过模型计算预测值
        loss = criterion(y_predict, target)     # 算损失
        loss.backward()		# 反向传播算法计算梯度
        optimizer.step()
        if index % 100 == 0:    # 每100次保存一次模型，打印报告
            torch.save(model.state_dict(), "./model/model.pkl")
            torch.save(optimizer.state_dict(), "./model/optimizer.pkl")
            print(loss.item())

def test():
    correct = 0
    total = 0
    with torch.no_grad():
        for data in test_loader:
            input, target = data
            output = model(input)
            _, predict = torch.max(output.data, dim=1)
            total += target.size(0)
            correct += (predict == target).sum().item()
    print(correct/total)


if __name__=='__main__':
    for i in range(10):
        train(i)
        test()
